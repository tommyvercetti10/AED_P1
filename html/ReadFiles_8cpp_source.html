161vector<Lesson> ReadFiles::sortLessonsByDay(vector<Lesson> lessons) {
162 vector<Lesson> sortedLessons = lessons;
163 auto customComparator = [](Lesson a, Lesson b) {
164 if ((int)a.getDay_() < (int)b.getDay_()) {
165 return true;
166 }
167 if ((int)a.getDay_() == (int)b.getDay_()) {
168 return a.getStartHour() < b.getStartHour();
169 }
170 return false;
171 };
172 sort(sortedLessons.begin(), sortedLessons.end(), customComparator);
173 return sortedLessons;
174}

184vector<Class> ReadFiles::getClassChances(std::string uc, std::string currentClass, Student student, bool addUC) {
185 vector<Class> classes_ = classes;
186 int year = currentClass[0] - '0';
187 if (addUC) year = getUC(uc).getYear();
188 // Remove the current class and all the classes from a year different than the current UC
189 classes_.erase(remove_if(classes_.begin(), classes_.end(), [currentClass, year](Class& elem) {
190 return elem.getClassCode() == currentClass || elem.getYear() != year;
191 }), classes_.end());
192 // Get the current schedule, removing the UC we want to trade
193 vector<Lesson> lessons = lessonsForStudent(student);
194 if (!addUC) {
195 lessons.erase(remove_if(lessons.begin(), lessons.end(), [currentClass, uc](Lesson& elem) {
196 return elem.getClasscode() == currentClass && elem.getUcCode() == uc;
197 }), lessons.end());
198 }
199 int i = 0;
200 bool overlap = false;
201 // Compare the current schedule with all the different classes of the UC
202 while (i < classes_.size()) {
203 vector<Lesson> classLessons = classes_[i].getLessons(uc);
204 if (classLessons.size() == 0) overlap = true;
205 for (int i = 0; i < lessons.size(); i++) {
206 for (int c = 0; c < classLessons.size(); c++) {
207 if (lessonsOverlap(lessons[i], classLessons[c])) {
208 overlap = true;
209 break;
210 }
211 }
212 if (overlap) break;
213 }
214 if (overlap)
215 {
216 classes_.erase(classes_.begin() + i);
217 overlap = false;
218 }
219 else i++;
220 }
221 i = 0;
222 // Check the balance
223 if (classes_.size() > 1) {
224 int max = max_(classes_);
225 int min = min_(classes_);
226 if (max - min > 4) {
227 while (i < classes_.size()) {
228 if (classes_[i].getLessons()[0].getStudents().size() != min)
229 classes_.erase(classes_.begin() + i);
230 else i++;
231 }
232 }
233 else if (max - min == 4) {
234 while (i < classes_.size()) {
235 if (classes_[i].getLessons()[0].getStudents().size() == max)
236 classes_.erase(classes_.begin() + i);
237 else i++;
238 }
239 }
240 }
241 // Check the occupation
242 int maxOccupation;
243 if (!classes_.empty()) maxOccupation = classes_[0].getLessons()[0].getMaxCap();
244 classes_.erase(remove_if(classes_.begin(), classes_.end(), [maxOccupation](Class& elem) {
245 return elem.getLessons()[0].getStudents().size() == maxOccupation;
246 }), classes_.end());
247 return classes_;
248}

413void ReadFiles::loadRequests() {
414 ifstream in("../log.txt");
415 string line, type, up, uc, oldClass, newClass, oldUc, newUc;
416 while(getline(in, line)) {
417 stringstream iss(line);
418 getline(iss, type, '-');
419 getline(iss, up, '-');
420 if (type == "ClassChange" || type == "UCAdd" || type == "UCRemove") getline(iss, uc, '-');
421 if (type == "ClassChange") getline(iss, oldClass, '-');
422 if (type == "UCChange") {
423 getline(iss, oldUc, '-');
424 getline(iss, oldClass, '-');
425 getline(iss, newUc, '-');
426 }
427 getline(iss, newClass, '\r');
428 if (type == "ClassChange") changeClass(getClass(oldClass), getClass(newClass), uc, getStudent(up), false);
429 if (type == "UCAdd") addUC(getUC(uc), getClass(newClass), getStudent(up), false);
430 if (type == "UCRemove") removeUC(getUC(uc), getClass(newClass), getStudent(up), false);
431 if (type == "UCChange") changeUC(getUC(oldUc), getClass(oldClass), getUC(newUc), getClass(newClass), getStudent(up), false);
432 }
